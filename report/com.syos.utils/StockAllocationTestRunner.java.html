public class StockAllocationTestRunner {



    public static void main(String[] args) {

        // Mock dependencies
        StockSelectionStrategy strategy = new FEFOStockSelectionStrategy(); // Replace with actual strategy implementation
        StoreDao storeDao = new StoreDaoImpl(); // Replace with actual StoreDao implementation
        ShelfDao shelfDao = new ShelfDaoImpl(); // Replace with actual ShelfDao implementation
        WebInventoryDao webInventoryDao = new WebInventoryDaoImpl(); // Replace with actual WebInventoryDao implementation

        String itemCode = "COCA-KOLA-1L";
        int requiredQuantity = 10;
        String shelfId = "TEST";

        AllocationRequest request = new AllocationRequest(itemCode, requiredQuantity, AllocationTarget.SHELF);
        request.setShelfId(shelfId);

        AllocationServiceImpl allocationService = new AllocationServiceImpl(strategy, storeDao, shelfDao, webInventoryDao);

        try {
            List<AllocatedRestockDTO> allocations;

            if (request.getTarget() == AllocationTarget.SHELF) {
                allocations = allocationService.allocateToShelf(request);
            } else {
                allocations = allocationService.allocateToWeb(request);
            }

            System.out.println("\n✅ Allocation Successful!");
            for (AllocatedRestockDTO dto : allocations) {
                System.out.printf("➡ Item: %-15s | Batch: %-10s | Qty Allocated: %d\n",
                        dto.getItemCode(),
                        dto.getBatchCode(),
                        dto.getAllocatedQuantity());
            }

        } catch (Exception e) {
            System.err.println("❌ Allocation Failed: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
